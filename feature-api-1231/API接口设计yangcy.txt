# API接口详细设计
#用户认证模块
#登录
post  /auth/login  #用户登录验证，接收工号和密码，验证用户身份
post  /auth/validate_credentials  #验证用户提供的工号和密码是否正确
put  /auth/password/update  #更新密码，修改密码后对原密码进行更新
post  /auth/logout  #退出登录

#普通用户登录
#设备权限填写
get  /user/devices  #获取当前用户关联的所有设备信息
get  /user/permission_forms/show  #显示用户关联的权限表单，可填写
post  /user/permission_forms/submit  #提交权限表单，进行设备权限信息数据库的更新

#管理员登录
#用户管理
get  /admin/users  #获取用户列表
post  /admin/users  #创建用户
get  /admin/users/{id}  #获取用户详情
put  /admin/users/{id}  #更新用户信息
delete  /admin/users/{id}  #删除用户
post  /admin/users/{id}/reset_password  #重置用户密码
#设备管理
post  /admin/devices/import  #导入设备管理清单
get  /admin/devices  #获取设备列表
put  /admin/devices/update_database  #更新设备管理数据库
#权限表单管理
get   /admin/permission_forms  #获取权限表单
post  /admin/permission_forms/{id}  #搜索信息，权限表单可进行条件筛选
post  /admin/permission_forms/export  #导出权限表单到excel
#安全检查管理
post  /admin/safety_checks/create_monthly  #创建安全检查
get   /admin/safety_checks/{check_id}/sampling_devices  #获取抽样设备,进行随机选择
post   /admin/safety_checks/{check_id}/fill_sampling_form  #填写抽样检查表
put  /admin/safety_checks/{check_id}/update_ sampling_form  #更新抽样检查表

#通用功能模块
post   /export/excel  #导出Excel文件
post   /data/paginate  #数据分页
get  /files/list/{file_type}  #文件列表,查看上传过的文件
